------------------------------------------------------------
						VIM
------------------------------------------------------------

-- VIM: PLUG (THE MANAGER) --
:PlugInstall		: Installs plugins listed in .vimrc
:PlugUpdate			: Updates all plugins
:PlugClean			: Removes plugins no longer in .vimrc

-- VIM: NAVIGATION & UI (Normal Mode) --
\h \j \k \l			: Jump Left, Down, Up, Right
\v					: Reloads your .vimrc configuration
\tf					: Toggle between current and last file you had open
Ctrl-W then =		: Make all splits equal size
; / ,				: Repeat f or t command forward/backward
+ / -         		: Resize window width (+5 / -5)
u					: Undo
Ctrl-R				: Redo
%					: Jump between matching pairs of brackets () or {}
gi					: Move to the last place you were in insert mode and enters insert mode
\cp / \cn			: Jump to previous/next CoC diagnostic (error/warning)
\fr                 : Find all references to the symbol/link under the cursor
\nf					: Shortcut for a new split. Just type the file path after
:Source				: Reloads your .vimrc configuration
:Shortcuts    		: Open this help file in a vertical split
:vsplit     	  	: Create vertical split
:q            		: Close current split

-- VIM: CODE EDITING & LSP (Plugin-Powered) --
\d					: Jump INTO Definition
\b					: Jump BACK from Definition 
\o					: Open Documentation for word under cursor (LSP)
\rn					: Rename variable project-wide (Global)
\ca					: Open CoC Code Actions (fixes/suggestions)
Tab					: Select CoC autocomplete suggestion
ci<char>            : Change Inside delimiter (e.g. ci" replaces text inside quotes)
di<char>            : Delete Inside delimiter (e.g. di" deletes text inside quotes)
gcc           		: Comment/Uncomment current line (Commentary plugin)
gc (in visual mode)	: Comment/Uncomment selection (Commentary plugin)
dd                  : Clear line contents (leaves empty line)
dl                  : Delete line entirely (default dd behavior)
space-J / space-K	: Add new line below / above without entering insert mode
ctrl-J / ctrl-K		: Move current line(or selected block) down / up 
*					: Highlight all occurrences of the word currently under the cursor
zc                  : Close the fold under the cursor
zo                  : Open the fold under the cursor
za                  : Toggle the fold under the cursor
zM                  : Close all folds in the file
zR                  : Open all folds in the file
:%s/old/new/gc      : Search and replace in whole file(adding c at the end asks confirmation on each match)
:copen :cclose		: Open/Close Error window
:cn :cp				: Jump to Next/Previous error
:Format				: Manually trigger LSP formatting

-- VIM: INSERT MODE SHORTCUTS --
qq                  : Exit insert mode (same as <Esc>)
;;                  : Append ; at end of line 
{{                  : Append { at end of line
,,                  : Insert , after next character
::                  : Append : at end of line

-- VIM: DEBUGGER --
\gb					: Toggle breakpoint
\gc					: Start / Continue debugging
\gs					: Stop debugging session
\gr					: Restart debugging
\gn	\gi \go			: Step over / Step into / Step out
\gq					: Reset Debugger
:InstallDebugger <adapter_name>

- VIM: MARKDOWN -
:Preview			: Open rendered preview of current file in terminal split (glow)
\mp					: Open rendered preview of current file in terminal split (glow)

------------------------------------------------------------
					LANGUAGE SPECIFIC 
------------------------------------------------------------

-- VIM: GO --
\r					: Go run
\t					: Go test
:GoAddTags			: Add JSON tags to struct fields

-- TERMINAL: GO --
go mod tidy 		: Clean up dependencies in go.mod/go.sum

------------------------------------------------------------
						TERMINAL
------------------------------------------------------------

-- TERMINAL: SHELL COMMANDS --
mv old new	    	: Rename or move a file/directory
cp old new			: Copy a file
rm file 	    	: Delete a file
Ctrl+R              : Reverse history search (type any part of a past command)
Ctrl+A / Ctrl+E     : Move cursor to beginning / end of line
Alt+B / Alt+F		: Move backward/forward one word
cd -                : Switch to previous working directory
sudo !!             : Re-run last command with sudo(!! repeats last command)
diff -u file1 file2	: Shows line-by-line differences between two files in unified format

-- TERMINAL: ALIAS --
shortcuts			: alias to 'cat ~/vimfiles/shortcuts.txt'

-- TERMINAL: RIPGREP --
rg -l '"http"'		: List files with literal "http" import
rg -t go "func"		: Search "func" only inside .go files
rg -i "text"		: Case-insensitive search
rg --hidden "str"	: Search including hidden files (like .vimrc)
rg -t sql "text"	: Search only inside .sql files

-- TERMINAL: GIT --
git commit --amend --no-edit
git config --global user.name "Vitor"
git config --global user.email "your@email.com"

-- TERMINAL: GITHUB CLI --
gh auth login		: Authenticate with your GitHub account
gh repo clone [repo]: Clone a repository locally
gh pr create		: Create a new pull request from current branch
gh pr list			: List open pull requests in current repo
gh issue list		: List open issues in the repository
gh browse			: Open the current repository in your browser
